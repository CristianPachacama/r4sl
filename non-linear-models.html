glm_prob_bands = exp(glm_se_bands) / (1 + exp(glm_se_bands))

## Loading required package: foreach

gam_fit = gam(wage ~ s(age, 4) + s(year, 4) + education, data = Wage)
par(mfrow = c(1, 3))
